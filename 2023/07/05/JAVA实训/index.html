ArrayList

含义

底层采用可变数组的线性列表

体系结构

  • List
    • ArrayList

特点

  • 元素有序,可以重复

    因为底层采用可变数组

  • 存放的是对象的引用

案例

public class ListDemo {
 
    //ArrayList
       //含义:底层采用可变数组的线性集合
       //体系结构
           //List接口
              //ArrayList类
       //注意点
           //集合里面存储的是对象引用 
       
    public static void main(String[] args) {
        User user = new User();
        user.name="zs";
        
        
        
        List<User> dataList= new ArrayList<>();
        
        //添加
        dataList.add(user);
        
        //获取
        System.out.println(dataList.get(0));
        
        user.name="ls";
        System.out.println(dataList.get(0));
        
        System.out.println(user.name);
        
        //常用的方法
           //add 添加
           //get 获取
           //isEmpty 是否为空
           //size 元素的个数
           //contains 是否包含
           //remove 删除
        
        List<String> nameList = new ArrayList<>();
        
        nameList.add("zs");
        nameList.add("ls");
        
        System.out.println(nameList.get(0));    //zs
        
        System.out.println(nameList.isEmpty()); //false
        
        System.out.println(nameList.size());    //2
        System.out.println(nameList.contains("ww"));  //false
        
        nameList.remove(0);
        
        System.out.println(nameList); //ls
    }
}
class User{
    
    public String name;

    @Override
    public String toString() {
        return "User [name=" + name + "]";
    }
}

public class FileOtherDemo {
    
    //罗列出D:\暨南大学\Day01文件夹下的所有视频名字
    
    //递归
       //含义:有限次数自己调用自己
       //注意点:递归次数【深度】太多【太深】会造成栈溢出[StackOverflowError]

        String parentPath = "D:\\暨南大学\\Day01";
        listVideoName(parentPath);
     
    }
    
    /**
     * 列出当前路径所在的视频名称和孩子文件夹的视频名称
     * @param parentPath
     */
    private static void listVideoName(String parentPath){
        
        if(parentPath==null || "".equals(parentPath.trim())){
            return;
        }
        
        //罗列出当前路径下的视频名称
        File parentFile = new File(parentPath);
        
        File children [] = parentFile.listFiles();
        
        for(File child : children){
            
            //判断是不是MP4视频格式
            if(child.isFile()){
                if(child.getName().endsWith(".mp4") ){
                    System.out.println(child.getName());
                }
            }else{
                //文件夹
                listVideoName(child.getAbsolutePath());
            }
            
        }
        
    }
}

代码统计

public class Test {
    
    static int sum=0;
    public static void main(String[] args) {
        
        String targetPath = "C:\\Users\\Administrator\\workspace";
        
        //统计多少行代码? 
           //思路
              //找到所有的.java文件
              //读取.java文件有多少行
              //累加
        
        listJavaFile(targetPath);
        System.out.println(sum);
    }
    
    /**
     * 递归读取java文件
     * @param parentPath
     */
    private static void listJavaFile(String parentPath){
        if(parentPath==null || "".equals(parentPath.trim())){
            return;
        }
        
        File parentFile = new File(parentPath);
        
        //罗列孩子
        File children [] = parentFile.listFiles();
        
        for(File child : children){
            
            if(child.isFile()){
                //是不是java文件
                if(child.getName().endsWith(".java")){
                    //是java文件 TODO
                    int count = calCount(child.getAbsolutePath());
                    sum+=count;
                }
            }else{
                //文件夹
                listJavaFile(child.getAbsolutePath());
            }
        }
    }


    /**
     * 读取某一个java文件的代码行数
     * @param path
     */
    private static int calCount(String path){
        int count = 0;
         
        BufferedReader bufferedReader=null;
        try {
            bufferedReader = new BufferedReader(new FileReader(path));
            
            //读一行加1
            String content = null;
            while((content=bufferedReader.readLine())!=null){
                if(!"".equals(content.trim())){
                    count++;
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        }finally {
            if(bufferedReader!=null){
                try {
                    bufferedReader.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
        return count;
    }
}

服务端

public class Server {
    
    public static void main(String[] args) {
        
        ServerSocket serverSocket = null;
        try {
            serverSocket = new ServerSocket(8080);
            
            System.out.println("等待客户端连接");
            
            //等待[阻塞]连接
            Socket socket = serverSocket.accept();
            
        
            System.out.println("有客户端连接了");
            //有客户端连接了
            
            //读取客户端发送过来的数据【输入流】
            InputStream inputStream = socket.getInputStream();
            
            byte [] cache = new byte[1024];
            int len = -1;
            
            StringBuilder content = new StringBuilder();
            //while((len=inputStream.read(cache))!=-1){
                len=inputStream.read(cache);
                content.append(new String(cache,0,len));
            //}
            
            System.out.println("收到客户端的内容"+content.toString());
            
            
            //响应数据给客户端【输出流】
            OutputStream outputStream = socket.getOutputStream();
            outputStream.write(content.toString().getBytes());
        } catch (IOException e) {
            e.printStackTrace();
        }finally {
            if(serverSocket!=null){
                try {
                    serverSocket.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
    }
}
